CREATE TABLE `customer_statement_transactions` (
    `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    `customer_statement_account_id` BIGINT UNSIGNED NOT NULL,
    `transaction_date` DATE NOT NULL,
    `ref_no` VARCHAR(255) NULL,
    `due_date` DATE NULL,
    `description` VARCHAR(255) NOT NULL,
    `debit` DECIMAL(15, 2) NOT NULL DEFAULT 0.00,
    `credit` DECIMAL(15, 2) NOT NULL DEFAULT 0.00,
    `balance` DECIMAL(15, 2) NOT NULL DEFAULT 0.00,
    `created_at` TIMESTAMP NULL,
    `updated_at` TIMESTAMP NULL,
    CONSTRAINT `cst_account_id_foreign` 
        FOREIGN KEY (`customer_statement_account_id`) 
        REFERENCES `customer_statement_accounts` (`id`) 
        ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

UPDATE customer_statement_transactions t
JOIN (
    SELECT
        t2.id,
        (
            a.beginning_balance +
            SUM(t2.debit - t2.credit) OVER (
                PARTITION BY t2.customer_statement_account_id
                ORDER BY t2.transaction_date, t2.id
                ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
            )
        ) AS correct_balance
    FROM customer_statement_transactions t2
    JOIN customer_statement_accounts a
        ON a.id = t2.customer_statement_account_id
) x ON x.id = t.id
SET t.balance = x.correct_balance;
